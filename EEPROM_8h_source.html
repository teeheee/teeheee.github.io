200 /***
201  EEPROMClass class.
202 
203  This object represents the entire EEPROM space.
204  It wraps the functionality of EEPtr and EERef into a basic interface.
205  This class is also 100% backwards compatible with earlier Arduino core releases.
206  ***/
213 struct EEPROMClass
214 {
215 
216  //Basic user access methods.
217  EERef operator[](const int idx)
218  {
219  return idx;
220  }
221  uint8_t read(int idx)
222  {
223  return EERef(idx);
224  }
225  void write(int idx, uint8_t val)
226  {
227  (EERef(idx)) = val;
228  }
229  void update(int idx, uint8_t val)
230  {
231  EERef(idx).update(val);
232  }
233 
234  //STL and C++11 iteration capability.
235  EEPtr begin()
236  {
237  return 0x00;
238  }
239  EEPtr end()
240  {
241  return length();
242  } //Standards requires this to be the item after the last valid entry. The returned pointer is invalid.
243  uint16_t length()
244  {
245  return E2END + 1;
246  }

248  //Functionality to 'get' and 'put' objects to and from EEPROM.
249  template<typename T> T &get(int idx, T &t)
250  {
251  EEPtr e = idx;
252  uint8_t *ptr = (uint8_t*) &t;
253  for (int count = sizeof(T); count; --count, ++e)
254  *ptr++ = *e;
255  return t;
256  }
257 
258  template<typename T> const T &put(int idx, const T &t)
259  {
260  EEPtr e = idx;
261  const uint8_t *ptr = (const uint8_t*) &t;
262  for (int count = sizeof(T); count; --count, ++e)
263  (*e).update(*ptr++);
264  return t;
265  }
266 };
