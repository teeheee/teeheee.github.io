130 template<class LinkType> boolean TPixy<LinkType>::getStart()
131 {
132  uint16_t w, lastw;
133 
134  lastw = 0xffff;
135  int timeout = 0;
136  while (true)
137  {
138  w = link.getWord();
139  if (w == 0 && lastw == 0)
140  {
141  delayMicroseconds(10);
142  //SERIAL_PRINTLN("nothing to be read");
143  return false;
144  }
145  else if (w == PIXY_START_WORD && lastw == PIXY_START_WORD)
146  {
147  blockType = NORMAL_BLOCK;
148  return true;
149  }
150  else if (w == PIXY_START_WORD_CC && lastw == PIXY_START_WORD)
151  {
152  blockType = CC_BLOCK;
153  return true;
154  }
155  else if (w == PIXY_START_WORDX)
156  {
157  //SERIAL_PRINT("reorder");
158  link.getByte(); // resync
159  }
160  lastw = w;
161  timeout++;
162  if(timeout>1000)
163  {
164  ERROR_MESSAGE("Pixy: i2c timeout");
165  return false;
166  }
167  }
168 }

176 template<class LinkType> uint16_t TPixy<LinkType>::getBlocks(uint16_t maxBlocks)
177 {
178  uint8_t i;
179  uint16_t w, checksum, sum;
180  Block *block;
181 
182  if (!skipStart)
183  {
184  if (getStart() == false)
185  return 0;
186  }
187  else
188  skipStart = false;
189 
190  for (blockCount = 0;
191  blockCount < maxBlocks && blockCount < PIXY_MAXIMUM_ARRAYSIZE;)
192  {
193  checksum = link.getWord();
194  if (checksum == PIXY_START_WORD) // we've reached the beginning of the next frame
195  {
196  skipStart = true;
197  blockType = NORMAL_BLOCK;
198  //SERIAL_PRINTLN("skip");
199  return blockCount;
200  }
201  else if (checksum == PIXY_START_WORD_CC)
202  {
203  skipStart = true;
204  blockType = CC_BLOCK;
205  return blockCount;
206  }
207  else if (checksum == 0)
208  return blockCount;
209 
210  if (blockCount > blockArraySize)
211  resize();
212 
213  block = blocks + blockCount;
214 
215 
216  for (i = 0, sum = 0; i < sizeof(Block) / sizeof(uint16_t); i++)
217  {
218  if (blockType == NORMAL_BLOCK && i >= 5) // skip
219  {
220  block->angle = 0;
221  break;
222  }
223  w = link.getWord();
224  sum += w;
225  *((uint16_t *) block + i) = w;
226  }
227 
228  if (checksum == sum)
229  blockCount++;
230  else{
231  //SERIAL_PRINT("cs error ");
232  }
233  w = link.getWord();
234  if (w == PIXY_START_WORD)
235  blockType = NORMAL_BLOCK;
236  else if (w == PIXY_START_WORD_CC)
237  blockType = CC_BLOCK;
238  else
239  return blockCount;
240  }
241  return 0;
242 }
